<script>
function showCategory(category) {
    let games = document.querySelectorAll(".game-card");

    games.forEach(game => {
        if (category === "all") {
            game.style.display = "block";
        } else {
            if (game.classList.contains(category)) {
                game.style.display = "block";
            } else {
                game.style.display = "none";
            }
        }
    });
}
</script>
